/* Criação de tabelas */

CREATE TABLE Funcionario (
    IdFuncionario INTEGER PRIMARY KEY,
    Nome VARCHAR,
    CPF VARCHAR,
    FK_Tanque_IdTanque INTEGER,
    FK_Turno_IdTurno INTEGER,
    FK_Cargo_IdCargo INTEGER
);

CREATE TABLE Turno (
    IdTurno INTEGER PRIMARY KEY,
    Horario VARCHAR
);

CREATE TABLE Cargo (
    IdCargo INTEGER PRIMARY KEY,
    NomeCargo VARCHAR
);

CREATE TABLE Contato (
    IdContato INTEGER PRIMARY KEY,
    Dado VARCHAR,
    FK_TipoContato_IdTipo INTEGER,
    FK_Funcionario_IdFuncionario INTEGER
);

CREATE TABLE TipoContato (
    IdTipo INTEGER PRIMARY KEY,
    Tipo VARCHAR
);

CREATE TABLE Endereco (
    IdFuncionario INTEGER PRIMARY KEY,
    Cidade VARCHAR,
    Bairro VARCHAR,
    Logradouro VARCHAR,
    Complemento VARCHAR,
    CEP VARCHAR,
    FK_Funcionario_IdFuncionario INTEGER
);

CREATE TABLE Tanque (
    IdTanque INTEGER PRIMARY KEY,
    Temperatura FLOAT,
    Ph FLOAT,
    Oxigenio FLOAT,
    FK_Peixe_IdPeixe INTEGER
);

CREATE TABLE Peixe (
    IdPeixe INTEGER PRIMARY KEY,
    NomeCientifico VARCHAR,
    TipoAgua BOOLEAN,
    TempMin FLOAT,
    TempMax FLOAT,
    PhMin FLOAT,
    PhMax FLOAT,
    OxMin FLOAT,
    OxMax FLOAT
);

CREATE TABLE HistoricoMedicao (
    IdHistorico INTEGER PRIMARY KEY,
    Temperatura FLOAT,
    Ph FLOAT,
    Oxigenio FLOAT,
    Hora TIME,
    Data DATE,
    FK_Tanque_IdTanque INTEGER
);
