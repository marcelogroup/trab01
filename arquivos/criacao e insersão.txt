CREATE TABLE Funcionario (
    IdFuncionario INTEGER PRIMARY KEY,
    Nome VARCHAR,
    CPF CHAR(12),
    IdTanque INTEGER,
    IdTurno INTEGER,
    IdCargo INTEGER
);

CREATE TABLE Turno (
    IdTurno INTEGER PRIMARY KEY,
    Horario VARCHAR
);

CREATE TABLE Cargo (
    IdCargo INTEGER PRIMARY KEY,
    NomeCargo VARCHAR
);

CREATE TABLE Contato (
    IdContato INTEGER PRIMARY KEY,
    Dado VARCHAR,
    IdTipo INTEGER,
    IdFuncionario INTEGER
);

CREATE TABLE TipoContato (
    IdTipo INTEGER PRIMARY KEY,
    Tipo VARCHAR
);

CREATE TABLE Endereco (
    Cidade VARCHAR,
    Bairro VARCHAR,
    Logradouro VARCHAR,
    Complemento VARCHAR,
    CEP VARCHAR,
    IdFuncionario INTEGER PRIMARY KEY
);

CREATE TABLE Tanque (
    IdTanque INTEGER PRIMARY KEY,
    IdPeixe INTEGER,
    QuatidadeSensor INTEGER,
    IdFuncionario INTEGER,
    Temperetura FLOAT,
    Ph FLOAT,
    Oxigenio FLOAT
);

CREATE TABLE Peixe (
    IdPeixe INTEGER PRIMARY KEY,
    Especie VARCHAR,
    TipoAgua VARCHAR,
    TempMin FLOAT,
    TempMax FLOAT,
    PhMin FLOAT,
    PhMax FLOAT,
    OxMin FLOAT,
    OxMax FLOAT
);

CREATE TABLE HistoricoMedicao (
    IdHistorico INTEGER PRIMARY KEY,
    Temperatura FLOAT,
    Ph FLOAT,
    Oxigenio FLOAT,
    IdTanque INTEGER,
    Hora TIME,
    Data DATE
);
